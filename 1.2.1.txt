import java.util.Scanner;

public class Main
{
    public static void main(String[] args)
    {
       Scanner in = new Scanner(System.in);
       System.out.println("Первый угол треугольника= ");
       double angle_1 = in.nextDouble();
       System.out.println("Второй угол треугольника= ");
       double angle_2 = in.nextDouble();
       String existence_of_a_triangle = (angle_2 + angle_1 < 180)?"Треугольник существует" : "Треугольник не существует";
System.out.println(existence_of_a_triangle);
}
}
